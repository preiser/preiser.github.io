What's so special about 3 and 5 anyway???

Pick your own fizzbuzz numbers, how very fun!

Here is the code for this silly project:

         
            var counter = 1;
            var string;
            var cloneDiv = $('#input').clone();

            $('#submit').click(function() {
               var usrchoice1 = document.getElementById("usrchoice1").value;
               var usrchoice2 = document.getElementById("usrchoice2").value;
               while (counter <= 100) {
                  string = "";
                  if (counter % usrchoice1 === 0 && counter % usrchoice2 === 0)
                  {
                     string = '<div class="fizzbuzz">fizzbuzz</div>';
                  } else if (counter % usrchoice1 === 0) {
                     string = '<div class="buzz">buzz</div>';
                  } else if (counter % usrchoice2 === 0) {
                     string = '<div class="fizz">fizz</div>';
                  } else {
                     string = '<div>' + counter + '</div>';
                  }
                  $('#usrNums').html('You chose ' + usrchoice1 + ' & ' + usrchoice2
                  + '.');
                  $('#fizzy').append(string);
                  counter++;
                  $('#input').replaceWith(cloneDiv);
               }
            });
